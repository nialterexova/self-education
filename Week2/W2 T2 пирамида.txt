#include "stdio.h"
#include "math.h" 
int main() {

float x = 0;
printf ("Введите высоту пирамиды: ");
scanf ("%f", & x); 

while (x<1 || x>23) {
printf ("Введите высоту пирамиды (от 1 до 23): ");
scanf ("%f", & x); }

for(int i = 0; i <= x; ++i) {
        
    for(int j = 0; j < x-i; ++j) {
      printf(" ");
    }
    for(int y = -2; y < i; ++y) {
      printf("#");
    }
    printf("\n");
  }
return 0;
}
